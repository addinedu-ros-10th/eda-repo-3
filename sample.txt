SELECT 
    COLUMN_NAME AS field_name,
    COLUMN_TYPE AS type,
    IS_NULLABLE,
    COLUMN_DEFAULT,
    COLUMN_COMMENT AS description
FROM INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'eda' AND TABLE_NAME = 'seoul_animal_hospital'
ORDER BY ORDINAL_POSITION;
